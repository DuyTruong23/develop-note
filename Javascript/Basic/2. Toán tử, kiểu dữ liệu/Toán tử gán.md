```jsx
/**
 Toán tử     Ví dụ     Tương đương
  =          x =y        x = y
  +=         x +=y       x = x + y
  -=         x -=y       x = x - y
  *=         x *=y       x = x * y
  /=         x /=y       x = x / y
  %=         x %=y       x = x % y
  **=        x **=y      x = x ** y
 */
var a = 1;
a += 2; //=> a = a + 2 => 3
a -= 1; //=> a = a - 1 => 2
a *= 2; //=> a = a * 2 => 4
a /= 2; //=> a = a / 2 => 2
a %= 2; //=> a = a % 2 => 0
a **= 2; //=> a = a ** 2 => 4
```
