```jsx
// +   --> Cộng
var a = 1;
var b = 2;
var c = a + b; //=> c = 3
// -   --> Trừ
var e = a - b; //=> e = -1
// *   --> Nhân
var d = a * b; //=> d = 2
// **  --> Lũy thừa
var f = a ** b; //=> f = 2
// /   --> Chia
var g = a / b; //=> g = 0.5
// %   --> Phần dư
var h = a % b; //=> h = 1
// ++  --> Tăng lên 1
var i = a++; //=> i = 1
// --  --> Giảm xuống 1
var j = a--; //=> j = 1
```
