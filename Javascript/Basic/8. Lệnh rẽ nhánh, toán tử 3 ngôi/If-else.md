```jsx
var date = 2;
//If -else
if (date === 2) {
  console.log('Today is Monday');
} else if (date === 3) {
  console.log('Today is Tuesday');
} else if (date === 4) {
  console.log('Today is Wednesday');
} else {
  console.log('Today is not a day');
}
// chỉ lọt vào 1 nhánh kiểm tra đúng, nếu tất cả đều sai thì lọt vào else
```
